## 4. 样题示例

### 4.1 L1样题

```rust
// 问题1: 修复编译错误
fn main() {
    let s1 = String::from("hello");
    let s2 = s1;
    println!("{}", s1); // 错误!
}

// 答案:
fn main() {
    let s1 = String::from("hello");
    let s2 = s1.clone(); // 或直接使用s1
    println!("{}", s1);
}

// 问题2: 实现安全函数
/// 返回切片中的最大值，如果为空返回None
fn find_max(arr: &[i32]) -> Option<i32> {
    // 实现
    arr.iter().copied().max()
}

// 问题3: 解释为什么这段代码是安全的
fn process(data: &[u8]) -> Vec<u8> {
    data.iter().map(|x| x * 2).collect()
}
// 答案: 输入只读，输出新建，无unsafe，边界检查由编译器保证
```

### 4.2 L2样题

```rust
// 问题1: 实现线程安全计数器
use std::sync::atomic::{AtomicU32, Ordering};

pub struct AtomicCounter {
    value: AtomicU32,
}

impl AtomicCounter {
    pub fn new() -> Self {
        Self {
            value: AtomicU32::new(0),
        }
    }

    pub fn increment(&self) {
        self.value.fetch_add(1, Ordering::Relaxed);
    }

    pub fn get(&self) -> u32 {
        self.value.load(Ordering::Relaxed)
    }
}

// 问题2: 设计ASIL B级状态机
// 要求: 无效状态转换必须在编译时阻止

pub struct Idle;
pub struct Running;
pub struct Error;

pub struct StateMachine<S> {
    _state: PhantomData<S>,
}

impl StateMachine<Idle> {
    pub fn new() -> Self {
        Self { _state: PhantomData }
    }

    pub fn start(self) -> StateMachine<Running> {
        StateMachine { _state: PhantomData }
    }
}

impl StateMachine<Running> {
    pub fn stop(self) -> StateMachine<Idle> {
        StateMachine { _state: PhantomData }
    }

    pub fn error(self) -> StateMachine<Error> {
        StateMachine { _state: PhantomData }
    }
}
```
