#### 1.1.2 泛型关联类型 (GATs)

**特性描述**:

```rust
trait Iterator {
    type Item<'a> where Self: 'a;
    
    fn next<'a>(&'a mut self) -> Option<Self::Item<'a>>;
}

struct MyIterator {
    data: Vec<i32>,
    index: usize,
}

impl Iterator for MyIterator {
    type Item<'a> = &'a i32;
    
    fn next<'a>(&'a mut self) -> Option<Self::Item<'a>> {
        if self.index < self.data.len() {
            let item = &self.data[self.index];
            self.index += 1;
            Some(item)
        } else {
            None
        }
    }
}
```
