    pub fn solve_gauss_elimination(&self) -> Result<Array1<f64>, String> {
        let mut augmented = self.matrix.clone();
        let mut b = self.vector.clone();
        let n = augmented.nrows();
        
        // å‰å‘æ¶ˆå…ƒ

                    augmented[[j, k]] -= factor * augmented[[i, k]];
