// ControlFlow 在迭代中的语义
fn find_negative(numbers: &[i32]) -> Option<i32> {
    numbers.iter().try_for_each(|&n| {
        if n < 0 { ControlFlow::Break(n) }
        else { ControlFlow::Continue(()) }
    }).break_value()
}

// 使用 array_windows 的算法实现
pub fn sliding_window_maximum(nums: &[i32], k: usize) -> Vec<i32> {
    nums.windows(k)  // 动态窗口
        .map(|window| window.iter().copied().max().unwrap_or(0))
        .collect()
}

pub fn array_window_pattern_match(data: &[u8]) -> bool {
    data.array_windows::<3>()  // 1.94 固定大小窗口
        .any(|[a, b, c]| a + b == *c)
}
