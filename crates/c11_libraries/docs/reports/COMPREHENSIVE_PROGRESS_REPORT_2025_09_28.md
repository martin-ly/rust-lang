#### 1.1 性能监控器修复

```rust
// 修复前：计算整个数组的最小值和最大值
pub fn get_min(&self) -> f64 {
    self.metrics.iter().fold(f64::INFINITY, |a, &b| a.min(b))
}

// 修复后：只计算实际记录的数据
pub fn get_min(&self) -> f64 {
    if self.total_samples == 0 {
        return 0.0;
    }
    
    let count = METRIC_COUNT.min(self.total_samples);
    self.metrics[..count].iter().fold(f64::INFINITY, |a, &b| a.min(b))
}
```
