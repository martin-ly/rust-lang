## 🚀 快速开始

### 5分钟快速体验

```rust
// 1. 条件语句
let score = 85;
let grade = if score >= 90 {
    "A"
} else if score >= 80 {
    "B"
} else {
    "C"
};

// 2. 模式匹配
match grade {
    "A" => println!("优秀!"),
    "B" => println!("良好!"),
    _ => println!("继续努力!"),
}

// 3. 闭包
let add = |x, y| x + y;
println!("结果: {}", add(3, 5));

// 4. 错误处理
fn divide(a: i32, b: i32) -> Result<i32, String> {
    if b == 0 {
        Err("除数不能为零".to_string())
    } else {
        Ok(a / b)
    }
}
```
